    <script>
        function drawLottery() {
            // Check if they already spun today
            let saved = localStorage.getItem('myDiscount');
            let win;

            if (saved) {
                win = saved;
            } else {
                // Probability: 70% get 10, 20% get 20, 10% get 30
                const rand = Math.random();
                if (rand < 0.7) win = "10%";
                else if (rand < 0.9) win = "20%";
                else win = "30%";
                
                localStorage.setItem('myDiscount', win);
            }

            document.getElementById('result').innerText = win;
            document.getElementById('spinBtn').style.display = 'none';
        }
    </script>
